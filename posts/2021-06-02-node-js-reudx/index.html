function reducer(todo = [], { type, text, id }) {
  switch (type) {
    case ACTIONS.ADD:
      return [{ id: uuidv4(), text }, ...todo];
    case ACTIONS.DELETE: {
      return todo.filter((aTodo) => aTodo.id !== id);
    }
    default:
      return todo;
  }
}
